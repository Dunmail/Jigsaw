Welcome to Jigsaw

Jigsaw is an open source framework based on the Interoperability Toolkit

Be productive

We've put together an easy to use framework to enable developers to get up to speed with ITK as rapidly as possible

Spin up an example ITK Host to test your client

var jigsaw = require('jigsaw');
var interactions = jigsaw.interactionHandler();
var process = function(req,res){
    console.log(req);
}
var routes = new Array();
routes.push(interactions.create("/docs", "requestException", [], process));
var app = jigsaw.createServer(routes);
app.addKey("client_public.pem");
app.listen(3000);

Need to test your own middleware - then use the jigsaw client

var client = require('jigsaw/client/client');
var envelope = require('jigsaw/client/messages/distributionEnvelope');
var msgProperties = {
    "serviceName" : "urn:nhs-itk:services:201005:SendDocument-v1-0",
    "key":  "client.pem",
    "properties": {"addresslist": new Array("urn:nhs-uk:addressing:ods:Y88764"),
                   "auditIdentity": new Array("urn:nhs-uk:addressing:ods:R59"),
                   "manifest": [
                      {"mimetype": "application/pdf", "data": "sampledata"}
                    ],
                    "senderAddress": "urn:nhs-uk:addressing:ods:R59:oncology"
                 },
    "url": "http://localhost:3000/docs",
    "handler": function(ctx) {
        console.log("your host sent this:" + ctx.response);
    } 
}
var msg = envelope.create(msgProperties);
client.send(msg);
